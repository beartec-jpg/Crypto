1. Create client/src/pages/CryptoChart.tsx
tsximport React from 'react';
import { useEffect } from 'react';

const CryptoChart: React.FC = () => {
  useEffect(() => {
    const script = document.createElement('script');
    script.type = 'text/javascript';
    script.src = 'https://s3.tradingview.com/tv.js';
    script.async = true;
    script.onload = () => {
      new window.TradingView.widget({
        width: '100%',
        height: 610,
        symbol: 'BINANCE:XRPUSDT',
        interval: '1D',
        timezone: 'Etc/UTC',
        theme: 'dark', // Matches indigo theme, can switch to 'light'
        style: '1',
        locale: 'en',
        toolbar_bg: '#1E3A8A', // Blue/indigo background
        enable_publishing: false,
        allow_symbol_change: true,
        container_id: 'tradingview_chart',
        // Custom indicators
        studies: [
          'RSI@tv-basicstudies',
          'MACD@tv-basicstudies',
          'MASimple@tv-basicstudies',
        ],
      });
    };
    document.body.appendChild(script);

    // Cleanup
    return () => {
      document.body.removeChild(script);
    };
  }, []);

  return (
    <div className="crypto-chart-container">
      <h1 className="text-2xl font-bold text-white mb-4" style={{ color: '#1E3A8A' }}>
        XRP/USDT Crypto Chart
      </h1>
      <div id="tradingview_chart" className="w-full h-full" />
    </div>
  );
};

export default CryptoChart;

tsximport React, { useEffect, useState } from 'react';

const CryptoChart: React.FC = () => {
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const script = document.createElement('script');
    script.type = 'text/javascript';
    script.src = 'https://s3.tradingview.com/tv.js';
    script.async = true;
    script.onload = () => {
      new window.TradingView.widget({
        width: '100%',
        height: 610,
        symbol: 'BINANCE:XRPUSDT',
        interval: '1D',
        timezone: 'Etc/UTC',
        theme: 'dark',
        style: '1',
        locale: 'en',
        toolbar_bg: '#1E3A8A',
        enable_publishing: false,
        allow_symbol_change: true,
        container_id: 'tradingview_chart',
        studies: ['RSI@tv-basicstudies', 'MACD@tv-basicstudies', 'MASimple@tv-basicstudies'],
      });
      setLoading(false);
    };
    script.onerror = () => setLoading(false);
    document.body.appendChild(script);

    return () => {
      document.body.removeChild(script);
    };
  }, []);

  return (
    <div className="crypto-chart-container">
      <h1 className="text-2xl font-bold text-white mb-4" style={{ color: '#1E3A8A' }}>
        XRP/USDT Crypto Chart
      </h1>
      {loading ? (
        <div>Loading chart...</div>
      ) : (
        <div id="tradingview_chart" className="w-full h-full" />
      )}
    </div>
  );
};

export default CryptoChart;
